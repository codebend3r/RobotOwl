function init() {

    var Crivas = {

        ViewModel: function (data) {

            var self = this;

            self.longList = data.portfolio;

            self.portfolioData = ko.observable(self.longList[0]);

            self.menuList = ko.observableArray(
                ko.utils.arrayMap(data.portfolio, function(i){
                    return {
                        id: i.id,
                        menuText: i.menutext,
                        slug: i.slug.toLowerCase()
                    };
                })
            );

            self.changePage = function(data) {
                self.portfolioData(self.longList[data.id]);
                //location.hash = 'portfolio/' + data.slug;
                Crivas.showNewSection();
            };

            self.model = this;

            return self;

        },

        cycleTime: 3000,

        cycleTimer: null,

        currentSlideNum: 0,

        numberOfPics: 0,

        currentSlideList: null,

        showNewSection: function() {

            console.log('showNewSection');

            Crivas.showPreloader();

            $('.stripedBorder').imagesLoaded(function(){

                console.log('ALL IMAGES LOADED ++');
                Crivas.allImagesLoaded();

            });
        },

        initCycle: function() {

            this.currentSlideList = $(".stripedBorder img");
            this.numberOfPics = this.currentSlideList.length - 1;
            this.currentSlideNum = this.currentSlideList.length - 1;

            clearInterval(this.cycleTimer);

            if (this.numberOfPics > 0) {
                this.cycleTimer = setInterval( this.cycleImages, this.cycleTime);
            }

        },

        /**
         Cycles through the next image in the array.
         If it's the last image in array it will reset back to the first one.

         @method cycleImages
         @return NULL
        **/
        cycleImages: function() {

            var currentSlide = Crivas.currentSlideList[ Crivas.currentSlideNum ];
            var firstSlide = Crivas.currentSlideList[ 0 ];
            var lastSlide = Crivas.currentSlideList[ Crivas.numberOfPics ];
            $(currentSlide).fadeOut( 300 );

            Crivas.currentSlideNum -= 1;

            if (Crivas.currentSlideNum <= -1) {
                Crivas.currentSlideNum = Crivas.numberOfPics;
                $(lastSlide).fadeIn( 300 );
                $(firstSlide).fadeIn( 300, Crivas.showAll );
            }

        },

        showAll: function() {
            $.each( Crivas.currentSlideList, function() {
                $(this).show();
            })
        },

        showPreloader: function() {
            $('.imagePreloader').show();
            $('.stripedBorder').hide();
        },

        allImagesLoaded: function() {
            $('.imagePreloader').hide();
            $('.stripedBorder').show();
            this.initCycle();
        },

        init: function() {
            var model = this.ViewModel(data);
            ko.applyBindings(model);

            Crivas.showNewSection();

        }

    }

    /*
    Sammy(function() {
        this.get('#/something/:section', function() {

        });

        this.set('#/something/:section', function() {

        });

    }).run();
    */


    Crivas.init();

    window.Site = Crivas;

};
